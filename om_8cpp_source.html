160 void model_parameters::initialize_model()
161 {
162  rt.initialize();
163  bt.initialize();
164  bo = mfexp(log_bo);
165  ro = bo*(1.-s);
166  reck = 4.*h/(1.-h);
167  a = reck*ro/bo;
168  b = (reck-1.0)/bo;
169  rt(syr,syr+agek) = ro * exp(wt(syr,syr+agek));
170  bt(syr) = bo;
171  sig = sqrt(1.0/mfexp(log_sigma));
172  tau = sqrt(1.0/mfexp(log_tau));
173 }

175 void model_parameters::population_dynamics()
176 {
177  int i;
178  dvariable btmp;
179  fpen.initialize();
180  for(i=syr;i<=nyr;i++)
181  {
182  ft(i) = -log((-ct(i)+bt(i))/bt(i));
183  if(i-syr > agek)
184  {
185  rt(i) = a*bt(i-agek)/(1.+b*bt(i-agek)) * exp(wt(i));
186  }
187  btmp = s*bt(i) + rt(i) - ct(i);
188  bt(i+1) = posfun(btmp,0.1,fpen);
189  }
190  sd_dep = bt(nyr)/bo;
191 }

221 void model_parameters::calc_objective_function()
222 {
228  nll.initialize();
229  dvariable isig2 = mfexp(log_sigma);
230  dvariable itau2 = mfexp(log_tau);
231  // No comments
232  nll(1) = dnorm(epsilon,sig);
233  nll(2) = dbeta(s,30.01,10.01);
234  nll(3) = dnorm(log_bo,log(3000),1.0);
235  nll(4) = dbeta((h-0.2)/0.8,1.01,1.01);
236  nll(5) = dgamma(isig2,1.01,1.01);
237  if(active(log_tau))
238  {
239  nll(6) = dgamma(itau2,1.01,1.01);
240  nll(7) = dnorm(wt,tau);
241  }
242  else
243  {
244  nll(7) = dnorm(wt,1.0);
245  }
246  if(fpen>0 && !mc_phase()) cout<<"Fpen = "<<fpen<<endl;
247  f = sum(nll) + 100000.*fpen;
248 }
