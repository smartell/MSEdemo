00145 void model_parameters::initialize_model(void)
00146 {
00147         bo               = mfexp(log_bo);
00148         ro               = bo*(1.-s);
00149         reck             = 4.*h/(1.-h);
00150         a                = reck*ro/bo;
00151         b                = (reck-1.0)/bo;
00152         rt(syr,syr+agek) = ro * exp(wt(syr,syr+agek));
00153         bt(syr)          = bo;
00154         sig              = sqrt(1.0/mfexp(log_sigma));
00155         tau              = sqrt(1.0/mfexp(log_tau));

00158 void model_parameters::population_dynamics(void)
00159 {
00160         int i;
00161         dvariable btmp;
00162         fpen.initialize();
00163         for(i=syr;i<=nyr;i++)
00164         {
00165                 ft(i) = -log((-ct(i)+bt(i))/bt(i));
00166                 if(i-syr > agek)
00167                 {
00168                         rt(i) = a*bt(i-agek)/(1.+b*bt(i-agek)) * exp(wt(i));    
00169                 }
00170                 btmp    = s*bt(i) + rt(i) - ct(i);
00171                 bt(i+1) = posfun(btmp,0.1,fpen);
00172         }
00173         sd_dep = bt(nyr)/bo;
00174 }

00183 void model_parameters::calc_objective_function(void)
00184 {
00185         nll.initialize();
00186         dvariable isig2 = mfexp(log_sigma);
00187         dvariable itau2 = mfexp(log_tau);
00188         // negative loglikelihoods and priors stored in nll vector
00189         nll(1) = dnorm(epsilon,sig);
00190         nll(2) = dbeta(s,30.01,10.01);
00191         nll(3) = dnorm(log_bo,log(3000),1.0);
00192         nll(4) = dbeta((h-0.2)/0.8,1.01,1.01);
00193         nll(5) = dgamma(isig2,1.01,1.01);
00194         if(active(log_tau))
00195         {
00196                 nll(6) = dgamma(itau2,1.01,1.01);
00197                 nll(7) = dnorm(wt,tau);
00198         }
00199         else
00200         {
00201                 nll(7) = dnorm(wt,1.0);
00202         }
00203         // objective function + penalty
00204         if(fpen>0 && !mc_phase()) cout<<"Fpen = "<<fpen<<endl;
00205         f = sum(nll) + 100000.*fpen;
00206 }
