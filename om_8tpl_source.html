00138 FUNCTION void initialize_model()
00139         rt.initialize();
00140         bt.initialize();
00141         bo               = mfexp(log_bo);
00142         ro               = bo*(1.-s);

00162 FUNCTION void population_dynamics()
00163         int i;
00164         dvariable btmp;
00165         fpen.initialize();
00166         for(i=syr;i<=nyr;i++)
00167         {
00168                 ft(i) = -log((-ct(i)+bt(i))/bt(i));
00169                 if(i-syr > agek)
00170                 {
00171                         rt(i) = a*bt(i-agek)/(1.+b*bt(i-agek)) * exp(wt(i));    
00172                 }
00173                 
00174                 btmp    = s*bt(i) + rt(i) - ct(i);
00175                 bt(i+1) = posfun(btmp,0.1,fpen);
